98 }

102 mesh.SetAllIndices(K_INDICES);

112 _topBorderWidth = resolvedStyle.borderBottomWidth != 0 ? resolvedStyle.borderBottomWidth : _topBorderWidth;

124 _topBorderWidth = resolvedStyle.borderTopWidth != 0 ? resolvedStyle.borderTopWidth : _topBorderWidth;

195 private Vector2 RotatePoint(Vector2 point, Vector2 pivot, float angle)
196 {
197 float rad = angle * Mathf.Deg2Rad;
198 float sin = Mathf.Sin(rad);
199 float cos = Mathf.Cos(rad);
200
201 Vector2 dir = point - pivot;
202
203 return new Vector2(
204 dir.x * cos - dir.y * sin + pivot.x,
205 dir.x * sin + dir.y * cos + pivot.y
206 );
207 }

213 public bool IsPointInTrapez()
214 {
215 Vector2 localPoint = GetVisualElementLocalCoordinates();
216 float width = resolvedStyle.width;
217 float height = resolvedStyle.height;
218 Vector2 pivot = new(width * 0.5f, height * 0.5f);
219
220 float topIndent = Mathf.Tan((180f - _topAngle) * 0.5f * Mathf.Deg2Rad) * height;
221
222 Vector2[] vertices =
223 {
224 RotatePoint(new Vector2(0, height), pivot, -_rotationAngle),
225 RotatePoint(new Vector2(topIndent, 0), pivot, -_rotationAngle),
226 RotatePoint(new Vector2(width - topIndent, 0), pivot, -_rotationAngle),
227 RotatePoint(new Vector2(width, height), pivot, -_rotationAngle)
228 };
229
230 bool inside = false;
231 for (int i = 0, j = vertices.Length - 1; i < vertices.Length; j = i++)
232 {
233 if ((vertices[i].y > localPoint.y) == vertices[j].y > localPoint.y) continue;
234
235 float intersectX = (vertices[j].x - vertices[i].x) * (localPoint.y - vertices[i].y) / (vertices[j].y - vertices[i].y) + vertices[i].x;
236 if (localPoint.x < intersectX)
237 {
238 inside = !inside;
239 }
240 }
241
242 return inside;
243 }
