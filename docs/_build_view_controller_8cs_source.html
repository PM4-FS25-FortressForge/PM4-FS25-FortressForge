180 [ServerRpc(RequireOwnership = false)]
181 private void TryPlaceBuildingServerRpc(int buildingIndex, HexTileCoordinate coord, float rotation)
182 {
183 BaseBuildingTemplate template = AvailableBuildings[buildingIndex];
184 (var shapeData, var isStackableList) = HexTileHelper.ExtractShapeInformation(template.ShapeDataEntries);
185
186 List<HexTileCoordinate> rotatedShape = HexTileHelper.GetRotatedShape(shapeData, rotation);
187 List<HexTileCoordinate> globalRotatedShape = rotatedShape.Select(tile => tile + coord).ToList();
188
189 HexGridData targetGrid = _ownedHexGridDatas
190 .FirstOrDefault(grid => grid.ValidateBuildingPlacement(coord, rotatedShape));
191
192 if (targetGrid == null || !targetGrid.EconomySystem.CheckForSufficientResources(template.GetBuildCost()))
193 {
194 Debug.Log("Server: Invalid placement or insufficient resources.");
195 return;
196 }
197
198 // Spawn the building prefab at the calculated position and rotation
199 Vector3 pos = coord.GetWorldPosition(_config.GridRadius, _config.TileHeight) +
200 HexTileHelper.GetAveragePosition(rotatedShape, _config.TileSize, _config.TileHeight);
201 Quaternion rot = Quaternion.Euler(0f, rotation, 0f) * template.BuildingPrefab.transform.rotation;
202 GameObject prefab = SpawnNetworked(template.BuildingPrefab, pos, rot, targetGrid.Id);
203 if (prefab == null)
204 {
205 Debug.LogError("Failed to spawn building prefab.");
206 return;
207 }
208 // Give ownership to the player who placed the building.
209 // Enforces that only the owner can interact with the placed buildings.
210 // Because of FishNet's ownership system, we need to do this directly inside the serverrpc method
211 NetworkObject netObj = prefab.GetComponent<NetworkObject>();
212 if (netObj != null && base.Owner.IsValid)
213 {
214 netObj.GiveOwnership(base.Owner);
215 }
216 if (template.GetType() == typeof(WeaponBuildingTemplate))
217 { // Quick n dirty init for prefab
218 prefab.GetComponent<WeaponInputHandler>().Init(targetGrid);
219 }
220
221 targetGrid.MarkBuildingTiles(coord, rotatedShape, isStackableList);
222 targetGrid.EconomySystem.PayResource(template.GetBuildCost());
223
224 // Add reference to the building manager for later use.
225 List<HexTileData> tileDatas = globalRotatedShape
226 .Select(coord => targetGrid.TileMap[coord])
227 .ToList();
228 targetGrid.BuildingManager.AddBuilding(new BuildingData(prefab, tileDatas, template));
229
230 SyncPlacedBuildingToClientsRpc(buildingIndex, coord, targetGrid.Id, rotation, prefab);
231 }

242 [ObserversRpc]
243 private void SyncPlacedBuildingToClientsRpc(int buildingIndex, HexTileCoordinate coord, int hexGridId,
244 float rotation,
245 GameObject prefab)
246 {
247 HexGridData targetGrid = _hexGridManager.AllGrids.FirstOrDefault(grid => grid.Id == hexGridId);
248 BaseBuildingTemplate template = AvailableBuildings[buildingIndex];
249 var (shapeData, isStackableList) = HexTileHelper.ExtractShapeInformation(template.ShapeDataEntries);
250 List<HexTileCoordinate> rotatedShape = HexTileHelper.GetRotatedShape(shapeData, rotation);
251 List<HexTileCoordinate> globalRotatedShape = rotatedShape.Select(tile => tile + coord).ToList();
252 targetGrid.MarkBuildingTiles(coord, rotatedShape, isStackableList);
253
254 // Add local reference to building manager for later use.
255 List<HexTileData> tileDatas = globalRotatedShape
256 .Select(coord => targetGrid.TileMap[coord])
257 .ToList();
258 var buildingData = new BuildingData(prefab, tileDatas, template);
259
260 var tileData = prefab.AddComponent<BuildingView>();
261 tileData.Init(buildingData, _config);
262
263 // Repeated steps from serverrpc dont need to be repeated server side
264 if (!IsServerInitialized)
265 {
266 targetGrid.BuildingManager.AddBuilding(buildingData);
267 targetGrid.MarkBuildingTiles(coord, rotatedShape, isStackableList);
268
269 if (template is WeaponBuildingTemplate)
270 {
271 var handler = prefab.GetComponent<WeaponInputHandler>();
272 if (handler != null)
273 handler.Init(targetGrid);
274 }
275 }
