150 private void AlignTabHeadersWithTrapezBorder()
151 {
152 VisualElement tabHeaders = _trapezElement.Q<TemplateContainer>()?
153 .Q<VisualElement>("building-selector-root")?
154 .Q<TabView>(className: "unity-tab-view")?
155 .Q<VisualElement>("unity-tab-view__header-container");
156
157 if (tabHeaders == null)
158 {
159 Debug.LogError("Tab headers not found.");
160 return;
161 }
162
163 List<VisualElement> tabHeaderList = tabHeaders.Children().ToList();
164
165 const float maxMargin = 40f;
166 const float minMargin = 4f;
167 int count = tabHeaderList.Count;
168
169 for (int i = 0; i < count; i++)
170 {
171 float margin = count > 1
172 ? Mathf.Lerp(maxMargin, minMargin, i / (float)(count - 1))
173 : maxMargin;
174 tabHeaderList[i].style.marginLeft = Length.Percent(margin);
175 }
176 }

231 private void BindItem(VisualElement item, int index)

237 item.RegisterCallback<PointerEnterEvent>(evt =>

242 float tooltipWidth = _tooltipContainer.resolvedStyle.width;
243 float tooltipHeight = _tooltipContainer.resolvedStyle.height;
244
245 _tooltipContainer.style.left = cardGlobalPos.x + (item.resolvedStyle.width - tooltipWidth) / 2;
246 _tooltipContainer.style.top = cardGlobalPos.y - tooltipHeight - 8f; // 8px Abstand nach oben
247
248 _tooltipContainer.style.visibility = Visibility.Visible;
249 });
250
251 item.RegisterCallback<PointerLeaveEvent>(_ => _tooltipContainer.style.visibility = Visibility.Hidden);
252
253 item.AddToClassList("building-card-template-container");
254
255 VisualElement cardFrame = item.Q<VisualElement>("building-card-frame");
256 VisualElement labelsContainer = cardFrame?.Q<VisualElement>("building-card-label-container");
257

262 }

264 labelsContainer.Q<Label>("building-card-name").text = building.name;
265 labelsContainer.Q<Label>("building-card-cost").text = building.name;

295 private string GetFullTooltipText(BaseBuildingTemplate building)
296 {
297 Dictionary<ResourceType, float> costs = building.GetBuildCost();
298 Dictionary<ResourceType, float> production = building.GetNetResourceChange();
299
300 string costText = costs.Count > 0
301 ? "Kosten:\n" + string.Join("\n", costs.Select(c => $"{c.Key}: -{c.Value}"))
302 : "Kosten:\nKeine";
303
304 string prodText = production.Count > 0
305 ? "Produktion:\n" + string.Join("\n", production.Select(p => $"{p.Key}: +{p.Value}"))
306 : "Produktion:\nKeine";
307
308 return $"{costText}\n\n{prodText}";
309 }
