168 private void PopulateTabView()
169 {
170 IEnumerable<VisualElement> tabContentList = _trapezElement.Q<TemplateContainer>()?
171 .Q<VisualElement>("weapon-selector-root")?
172 .Q<TabView>(className: "unity-tab-view")?
173 .Q<VisualElement>("unity-tab-view__content-container")?
174 .Children();
175
176 if (tabContentList == null)
177 {
178 Debug.LogError("Tab content containers not found.");
179 return;
180 }
181
182 var groupedLabels = _weaponLabels
183 .GroupBy(label =>
184 {
185 var container = WeaponBuildingManager.Instance
186 .GetType()
187 .GetField("_weaponBuildings", System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance)
188 .GetValue(WeaponBuildingManager.Instance) as IEnumerable<object>;
189 var weaponClass = WeaponClasses.Artillery;
190 foreach (var c in container)
191 {
192 var labelField = c.GetType().GetProperty("Label").GetValue(c) as Label;
193 if (labelField == label)
194 {
195 weaponClass = (WeaponClasses)c.GetType().GetProperty("WeaponClass").GetValue(c);
196 break;
197 }
198 }
199
200 return weaponClass;
201 });
202
203 foreach (VisualElement tabContent in tabContentList)
204 {
205 if (!System.Enum.TryParse(tabContent.name, out WeaponClasses weaponClass))
206 continue;
207
208 var labelsForClass = groupedLabels.FirstOrDefault(g => g.Key == weaponClass);
209 if (labelsForClass == null)
210 continue;
211
212 VisualElement tabContentListView = tabContent.Q<VisualElement>("unity-tab__content-container")?.Q<ScrollView>();
213 if (tabContentListView == null)
214 {
215 Debug.LogError("Tab content list view not found.");
216 continue;
217 }
218
219 tabContentListView.Clear();
220
221 var labelList = labelsForClass.ToList();
222 int labelCount = labelList.Count;
223 const float maxMargin = 20f;
224 const float minMargin = 0f;
225
226 for (int i = 0; i < labelCount; i++)
227 {
228 float margin = labelCount > 1
229 ? Mathf.Lerp(maxMargin, minMargin, i / (float)(labelCount - 1))
230 : maxMargin;
231
232 Label weaponLabel = labelList[i];
233 weaponLabel.style.marginLeft = Length.Percent(margin);
234 weaponLabel.AddToClassList("weapon-label");
235 tabContentListView.Add(weaponLabel);
236 AddOnClickToWeaponLabels(weaponLabel);
237 }
238 }

291 private void SelectNextWeaponLabel()
292 {
293 if (_weaponLabels.Count == 0 || _bottomFrame.parent.parent.resolvedStyle.display == DisplayStyle.None) return;
294
295 int currentIndex = _selectedWeaponLabel != null ? _weaponLabels.IndexOf(_selectedWeaponLabel) : -1;
296 int nextIndex = (currentIndex + 1) % _weaponLabels.Count;
297
298 SelectWeaponLabel(_weaponLabels[nextIndex]);
299 WeaponBuildingManager.Instance.SelectWeaponBuildingByLabel(_weaponLabels[nextIndex]);
