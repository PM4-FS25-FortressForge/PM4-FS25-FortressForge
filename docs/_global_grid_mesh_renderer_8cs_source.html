52 float tileRadius = _gameStartConfiguration.TileSize;
53 float tileHeight = _gameStartConfiguration.TileHeight;
54 var terrainSize = _gameStartConfiguration.Terrain.terrainData.size;
55 float maxX = terrainSize.x;
56 float maxZ = terrainSize.z;
57
58 // Precompute unit circle directions for hex corners
59 Vector3[] hexDirs = new Vector3[6];
60 for (int i = 0; i < 6; i++)
61 {
62 float angle = Mathf.Deg2Rad * (60f * i);
63 hexDirs[i] = new Vector3(Mathf.Cos(angle), 0, Mathf.Sin(angle));
64 }
65
66 // Estimate the grid bounds in axial coordinate space
67 int maxQ = Mathf.CeilToInt(maxX / (tileRadius * Mathf.Sqrt(3f)));
68 int maxR = Mathf.CeilToInt(maxZ / (tileRadius * 1.5f));
69
70 // Accumulators for building a single wireframe mesh
71 List<Vector3> allVertices = new();
72 List<int> allIndices = new();
73 int vertexOffset = 0;
74
75 // Iterate all hex coordinates within estimated bounds
76 for (int r = -maxR; r <= maxR; r++)
77 {
78 for (int q = -maxQ; q <= maxQ; q++)
79 {
80 var coord = new HexTileCoordinate(q, r, 0);
81 Vector3 basePos = coord.GetWorldPosition(tileRadius, tileHeight);
82
83 // Clip tiles outside terrain bounds
84 if (basePos.x < 0 || basePos.x > maxX || basePos.z < 0 || basePos.z > maxZ)
85 continue;
86
87 // Sample terrain height at the tile center
88 basePos.y = _terrainHeightProvider.SampleHexHeight(basePos, tileHeight, tileRadius);
89
90 vertexOffset = CreateRenderTile(basePos, hexDirs, tileRadius, allVertices, allIndices, vertexOffset);
91 }
92 }
93
94 // If no tiles were generated, print a warning
95 if (allVertices.Count == 0)
96 {
97 Debug.LogWarning("No hex edges generated. Check tile size or terrain bounds.");
98 return;
99 }
100
101 // Create and assign the mesh
102 var mesh = new Mesh
103 {
104 name = "HexEdgeGrid",
105 indexFormat = UnityEngine.Rendering.IndexFormat.UInt32 // Support >65k vertices
106 };
107
108 mesh.SetVertices(allVertices);
109 mesh.SetIndices(allIndices, MeshTopology.Lines, 0);
110 mesh.RecalculateBounds();
111
112 _meshFilter.mesh = mesh;
113 gameObject.isStatic = true; // Enable static batching (optional)
114
115 Debug.Log($"Hex grid mesh generated in {sw.Elapsed.Ticks / 10000} ms");
116 }

121 private static int CreateRenderTile(Vector3 basePos, Vector3[] hexDirs, float tileRadius, List<Vector3> allVertices, List<int> allIndices, int vertexOffset)
122 {
123 // Generate edges for each side of the hex tile
124 for (int i = 0; i < 3; i++)

131 allIndices.Add(vertexOffset++);
