122 private static Dictionary<ResourceType, float> StabilizeEconomy(ref Dictionary<ResourceType, float> newResources, ref List<(IEconomyActor, Dictionary<ResourceType, float>)> resourceChanges)
123 {
124 do
125 {
126 foreach (ResourceType resourceType in AllResourceTypes)
127 {
128 if (newResources[resourceType] >= 0) continue;
129
130 var consumersAscendingByConsumption = resourceChanges
131 .Where(rc => rc.Item2.ContainsKey(resourceType) && rc.Item2[resourceType] < 0)
132 .OrderBy(rc => rc.Item2.ContainsKey(resourceType) ? rc.Item2[resourceType] : 0)
133 .ToList();
134
135 foreach (var consumer in consumersAscendingByConsumption)
136 {
137 if (newResources[resourceType] >= 0) break;
138
139 // Disable must disable or reduce building drain
140 consumer.Item1.Disable();
141
142 resourceChanges.Remove(consumer);
143 Debug.Log($"Disabled {consumer.Item1} due to negative {resourceType} balance.");
144
145 foreach (var resourceFromChange in consumer.Item2)
146 {
147 newResources[resourceFromChange.Key] -= resourceFromChange.Value;
148 }
149 }
150 }
151 } while (newResources.Any(r => r.Value < 0));
152
153 return newResources;
154 }
