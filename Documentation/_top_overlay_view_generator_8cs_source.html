34 public void Init(EconomySync economySync)
35 {
36 if (economySync == null)
37 {
38 Debug.LogError("EconomySync is null in TopOverlayViewGenerator!");
39 return;
40 }
41
42 economySync.SyncedResources.OnChange += (op, key, value, asServer) =>
43 {
44 if (op != SyncDictionaryOperation.Set) return; // Only update on set operation
45 if (key == ResourceType.GlobalMagma)
46 {
47 UpdatePlanetMagmaFillableContainer(value);
48 return;
49 }
50
51 if (_fillableRessourceContainers.TryGetValue(key, out var container))
52 {
53 UpdateRessourceFillableContainer(_overlayFrame, container, value);
54 }
55 };
56
57 // Optionally preload initial values
58 foreach (var entry in economySync.SyncedResources)
59 {
60 if (entry.Key == ResourceType.GlobalMagma)
61 {
62 UpdatePlanetMagmaFillableContainer(entry.Value);
63 break;
64 }
65
66 if (_fillableRessourceContainers.TryGetValue(entry.Key, out var container))
67 {
68 UpdateRessourceFillableContainer(_overlayFrame, container, entry.Value);
69 }
70 }
