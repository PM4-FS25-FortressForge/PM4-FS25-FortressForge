126 {

148 public HexTileData CreateOrClaimHexTile(HexTileCoordinate newHexCoords)
149 {
150 // Find if tile is in another grid already
151 HexGridData previousOwner = _hexGridManager.AllGrids
152 .FirstOrDefault(grid => grid.TileMap.ContainsKey(newHexCoords));
153
154 if (previousOwner == this)
155 return TileMap[newHexCoords];
156
157 // If the tile is not owned by any grid, create a new tile.
158 if (previousOwner == null)
159 return CreateTile(newHexCoords);
160
161 // Check if the owner is the independent grid, in this case, we can just claim it.
162 if (previousOwner == _hexGridManager.IndependentGrid)
163 return ClaimTile(newHexCoords);
164
165 // If the tile is owned by another grid, do nothing and return null.
166 Debug.LogWarning("Tile already owned by another grid. Cannot create new tile.");
167 return null;
168 }

187 public HexTileData CreateTile(HexTileCoordinate newHexCoords)
188 {
189 if (!ValidateTilePlacement(newHexCoords))
190 return null;
191
192 var tile = new HexTileData(newHexCoords)
193 {
194 IsOwned = IsOwned,
195 IsInvisible = IsInvisible,
196 };
197
198 // Create and store the delegate
199 tile.OnHoverChanged += tileData => OnHoverTileChanged?.Invoke(tileData);
200 tile.OnChanged += hexTileData => OnChanged?.Invoke(hexTileData);
201 OnNewTileCreated?.Invoke(tile, newHexCoords);
202 TileMap[newHexCoords] = tile;
203
204 return tile;
205 }

268 public bool ValidateTilePlacement(HexTileCoordinate hexCoord)
269 {
270 // Check if lowest on the grid
271 var lowestTile = _terrainHeightProvider.GetHexTileCoordinate(
272 hexCoord.GetWorldPosition(TileRadius, TileHeight),
273 TileHeight, TileRadius);
274 if (lowestTile.H != hexCoord.H)
275 {
276 // Independent grid are not allowed to be placed on top of other grids
277 if (IsIndependent)
278 return false;
279
280 // Check if tile below
281 hexCoord.H -= 1;
282 return TileMap.ContainsKey(hexCoord);
283 }
284
285 return true;
286 }
