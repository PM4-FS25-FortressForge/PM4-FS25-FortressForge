140 private void HandleMovement(float deltaTime)
141 {
142 Vector2 moveInput = _moveTargetAction.ReadValue<Vector2>(); // WASD input
143 Vector3 moveDir = new Vector3(moveInput.x, 0, moveInput.y); // X (A/D), Z (W/S)
144 Vector3 moveVector = Quaternion.Euler(0, Yaw, 0) * moveDir; // Move relative to current yaw
145 TargetPosition += moveVector * MoveSpeed * deltaTime * _moveSpeedZoomSensitivity; // calculate new target position
146 }
147
154 private void HandleRotation(float deltaTime)
155 {
156 float rotateInput = _rotateAction.ReadValue<float>(); // Q/E input
157 Yaw = (Yaw + rotateInput * RotationSpeed * deltaTime * _pitchAndRollSpeedZoomSensitivity) % 360f; // Rotate around target
158 }
159
166 private void HandlePitch(float deltaTime)
167 {
168 float pitchInput = _pitchAction.ReadValue<float>(); // Up/Down arrows
169 Pitch += pitchInput * PitchSpeed * deltaTime * _pitchAndRollSpeedZoomSensitivity; // move camera in pitch angle to center
170 Pitch = Mathf.Clamp(Pitch, PitchLimits.y, PitchLimits.x); // Limit pitch angle
171 }
172
179 private void HandleZoom(float deltaTime)
180 {
181 if (UIClickChecker.Instance.IsMouseOnOverlay()) return;
182 float zoomInput = _zoomAction.ReadValue<float>(); // Zoom input with mouse wheel (made more smooth with SmoothDamp)
183 _targetZoom = Mathf.Clamp(_targetZoom - zoomInput * MouseWheelZoomSpeed, zoomLimits.x, zoomLimits.y);
184
185 float zoomButtonInput = _zoomButtons.ReadValue<float>(); // Zoom input with the Buttons
186 _targetZoom = Mathf.Clamp(_targetZoom - zoomButtonInput * ButtonsZoomSpeed * deltaTime, zoomLimits.x, zoomLimits.y);
187
188 Zoom = Mathf.SmoothDamp(Zoom, _targetZoom, ref _zoomVelocity, _zoomSmoothTime); // SmoothDamp for zooming smoothly in/out
189
190 // Linear mapping from [min, max Zoom] to [0.4, 3] for WASD movement
191 _moveSpeedZoomSensitivity = _moveSensitivityLimits.x + (Zoom - zoomLimits.x) * (_moveSensitivityLimits.y - _moveSensitivityLimits.x) / (zoomLimits.y - zoomLimits.x);
192 // Linear mapping from [min, max Zoom] to [0.85, 1.5] for pitch and roll
193 _pitchAndRollSpeedZoomSensitivity = _pitchAndRollSensitivityLimits.x + (Zoom - zoomLimits.x) * (_pitchAndRollSensitivityLimits.y - _pitchAndRollSensitivityLimits.x) / (zoomLimits.y - zoomLimits.x);
194 }

202 private void UpdateCameraPosition()
203 {
204 Quaternion rotation = Quaternion.Euler(Pitch, Yaw, 0); // Calculate the rotation around the centred object
205 Vector3 offset = rotation * new Vector3(0, 0, -Zoom); // Calculate the offset of the camera
206
207 TargetPosition.y = GetTerrainHeight(TargetPosition); // get the height of the terrain at the current position
208 _targetHeight = Mathf.SmoothDamp(_targetHeight, TargetPosition.y, ref _heightVelocity, _heightSmoothTime); // SmoothDamp for height
209 TargetPosition.y = _targetHeight;
210
211 transform.position = TargetPosition + offset; // Set the new position of the camera
212 transform.LookAt(TargetPosition); // Always look at the center point
213 }
