226 [ServerRpc(RequireOwnership = false)]
227 private void AdjustCannonAngle(float angleInput, float deltaTime)
228 {
229 // current rotation
230 Vector3 currentRotation = _cannonShaft.localEulerAngles;
231
232 // Convert to signed angle (-180 to 180)
233 float currentPitch = currentRotation.x;
234
235 if (currentPitch > 180f) currentPitch -= 360f;
236
237 // Calculate new pitch
238 float newPitch = currentPitch + angleInput * _constants.pitchSpeed * deltaTime;
239
240 // Clamp angle
241 newPitch = Mathf.Clamp(newPitch, _constants.minCannonAngle, _constants.maxCannonAngle);
242
243 // Apply adjusted angle
244 _cannonShaft.localEulerAngles = new Vector3(newPitch, currentRotation.x, currentRotation.z);
245 }

269 private void FireWeaponInternal()
270 {
271 switch (true)
272 {
273 case true when !_isAutoFiring && !_isReloading && !_isOnCooldown && _currentAmmo > 0 && _autoFireCoroutine == null: // Fire if not already firing and reloaded
274 _isAutoFiring = true;
275 Debug.Log("Firing");
276 _autoFireCoroutine = StartCoroutine(AutoFire());
277 break;
278
279 case true
280 when _isAutoFiring && _currentAmmo > 0 && _autoFireCoroutine != null: // Denies the player to fire while already firing
281 Debug.Log("Already firing");
282 break;
283
284 case true
285 when !_isReloading && _currentAmmo <= 0 && _reloadCoroutine == null: // Reload if not already reloading and out of ammo
286 ReloadWeaponServerRpc();
287 break;
288
289 case true
290 when _isReloading && _currentAmmo <= 0 && _reloadCoroutine != null: // Denies the player to reload while already reloading
291 Debug.Log("Already reloading");
292 break;
293 }
294 }

299 private IEnumerator AutoFire()

366 if (netObj != null)
