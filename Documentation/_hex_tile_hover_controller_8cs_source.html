63 public HexTileData GetCurrentlyHoveredTile()
64 {
65 if (_currentlyHoveredTile != null)
66 {
67 return _currentlyHoveredTile;
68 }
69 var ray = _mainCamera.ScreenPointToRay(Input.mousePosition);
70 if (!Physics.Raycast(ray, out RaycastHit hit))
71 return null;
72
73 var tileSize = _gameStartConfiguration.TileSize;
74 var tileHeight = _gameStartConfiguration.TileHeight;
75
76 Plane horizontalPlane = new Plane(Vector3.up, new Vector3(0, new HexTileCoordinate(tileSize, tileHeight, hit.point).GetWorldPosition(tileSize,tileHeight).y, 0));
77 if (!horizontalPlane.Raycast(ray, out float enter))
78 return null;
79
80 HexTileCoordinate coord = _terrainHeightProvider.GetHexTileCoordinate(ray.GetPoint(enter), tileHeight, tileSize);
81
82 var newHexTile = _hexGridManager.GetHexTileDataOrCreate(coord);
83
84 return newHexTile;
